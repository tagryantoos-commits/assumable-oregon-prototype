	var SessionTimeOutWarning = function (ml, fTouchSession, strLogoutLoc, dom, strLoginLoc) {
		this.m_strSessionExpireMsg = ml.SessionExpiredMsg;
		this.m_strMLMinute = ml.SessionExpireMinute;
		this.m_fMLMinutes = ml.SessionExpireMinutes;

		this.m_$warnDiv = $("#" + dom.Warning);
		this.m_$errorDiv = $("#" + dom.Error);

		this.m_strLocalStorageKey = 'matrix-session-channel';
		this.touchSession = fTouchSession;
		this.m_strLogoutLocation = strLogoutLoc;
		this.m_strLoginLocation = strLoginLoc;

		var strGeneratedId = '';
		var strValidCharacters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';

		for (var i = 0; i < 10; i++) {
			strGeneratedId += strValidCharacters.charAt(Math.floor(Math.random() * strValidCharacters.length));
		}

		this.m_strSignalerId = strGeneratedId;

		//bind the storage event handler - fired when localstorage changes
		if (!!window.localStorage) {
			var t = this;

			$(window).on('storage.sessiontimeout', function (e) {
				var evt = e.originalEvent;

				if (evt.key == t.m_strLocalStorageKey) {
					t.HandleStorageEvent(JSON.parse(evt.newValue));
				}
			});
		}
	};

	SessionTimeOutWarning.prototype.FireStorageEvent = function (event, data) {
		if (!!localStorage && !!window.JSON) {
			localStorage.setItem(this.m_strLocalStorageKey, JSON.stringify({
				'strSignalerId': this.m_strSignalerId,
				'event': event,
				'event_data': data,
				'event_timeStamp': Math.round(new Date().getTime() / 1000)
			}));
		}
	};

	SessionTimeOutWarning.prototype.HandleStorageEvent = function (evt) {
		if (typeof (evt) !== 'undefined') {
			switch (evt.event) {
				case 'rebind':
					this.BindTimers(evt);
					break;
				case 'abandon':
					this.SessionOver(evt);
					break;
			}
		}
		else {
			return false;
		}
	};

	SessionTimeOutWarning.prototype.RefreshTimers = function (nTimeout, nWarning) {
		this.RefreshSession(nTimeout, nWarning);
		this.FireStorageEvent('rebind', nTimeout + "|" + nWarning);
	};

	SessionTimeOutWarning.prototype.BindTimers = function (evt) {
		if (evt && evt.strSignalerId != this.m_strSignalerId) {
			var data = evt.event_data.split('|');
			if (data.length == 2) {
				this.RefreshSession(parseInt(data[0], 10), parseInt(data[1], 10));
			}
		}
	};

	SessionTimeOutWarning.prototype.SessionOver = function (evt) {
		if (!evt || evt.strSignalerId != this.m_strSignalerId) {
			this.HideMessage();
			this.m_$errorDiv.show();
		}
	};

	SessionTimeOutWarning.prototype.RefreshSession = function (nTimeout, nWarning) {
		localStorage.setItem('matrix-session-land', new Date().toGMTString());
		this.HideMessage();
		window.clearTimeout(this.m_oCheckHandle);
		window.clearTimeout(this.m_oWarnHandle);

		var self = this;
		this.m_oCheckHandle = window.setTimeout(function () {
			self.SessionOver.call(self);
		}, nTimeout * 60 * 1000 - 20 * 1000); //show session-over 20s before the session actually dies
		if (nWarning > 0) {
			if (nTimeout - nWarning > 0) {
				this.m_oWarnHandle = window.setTimeout(function () {
					self.ShowWarning.call(self, nTimeout);
				}, (nTimeout - nWarning) * 60 * 1000);
			}
			else {
				this.ShowWarning(nTimeout);
			}
		}
	};

	SessionTimeOutWarning.prototype.ShowWarning = function (nSessionMaxLength) {
		var dtStoredStamp = new Date(localStorage.getItem('matrix-session-land'));
		var self = this;
		var fUpdate = function () {
			var dtCurrentTime = new Date();
			var nTimeLeft = (dtStoredStamp.getTime() + parseInt(nSessionMaxLength * 60000, 10)) - dtCurrentTime.getTime();
			var nMinLeft = Math.round((nTimeLeft / 60000));

			var strMessage = nMinLeft === 1 ? self.m_strMLMinute : self.m_fMLMinutes({ expiremin: nMinLeft });
			if (nMinLeft > 0) {
				self.m_$warnDiv.find('span.message').text(strMessage);
			}
		};
		fUpdate();
		this.m_oWarnInterval = setInterval(fUpdate, 20 * 1000);
		this.m_$warnDiv.show();
	};

	SessionTimeOutWarning.prototype.HideMessage = function () {
		clearInterval(this.m_oWarnInterval);
		this.m_$warnDiv.find('span.message').text('');
		this.m_$warnDiv[0].style.display = "none";
	};

	SessionTimeOutWarning.prototype.PokeSession = function () {
		this.touchSession();
	};

	SessionTimeOutWarning.prototype.LogOut = function () {
		this.FireStorageEvent('abandon', '');
		if (!!this.m_strLogoutLocation) {
			window.location.replace(this.m_strLogoutLocation);
		}
	};

	SessionTimeOutWarning.prototype.Login = function () {
		if (!!this.m_strLoginLocation) {
			window.location.replace(this.m_strLoginLocation);
		}
	};

	$('#confirmationContainer').on('mtrx.confirmation', function (ev, data)
	{
		function kind2alert(kind)
		{
			switch (kind)
			{
				case 0: return 'primary';
				case 1: return 'success';
				case 2: return 'warning';
				case 3: return 'danger';
			}
		}
		function kindoficon(kind)
		{
			switch (kind)
			{
				case 0: return '';
				case 1: return '<i class="h3 far fa-check-circle"></i>';
				case 2: return '<i class="h3 far fa-exclamation-triangle"></i>';
				case 3: return '<i class="h3 far fa-exclamation-circle"></i>';
			}
		}
		var conf = $('#confirmationTemplate').html();
		const kind = kind2alert(data.Kind);
		conf = conf.replace("{kind}", kind)
			.replace("{message}", data.Message)
			.replace("{view}", data.View ? '<a href="'+data.View+'">'+data.ViewText+'</a>' : '')
			.replace("{undo}", data.Undo ? '<a href="' + data.Undo + '">' + data.UndoText + '</a>' : '')
			.replace("{icon}", kindoficon(data.Kind))
			.replace("{iconclass}", kind2alert(data.Kind) == 'primary' ? '' : 'mtrx-alert--withIcon')
		$('#confirmationContainer').html(conf);
		if(data.autoCloseForAll)
		{
			setTimeout(() => {$(`.alert-${kind}`).alert('close');}, data.Duration || 10000);
		}
		else
		{
			setTimeout(() => {$(".alert-success").alert('close');}, data.Duration || 10000);
		}
	});

		//****************************************************************************
		//	Show selected news item details in a modal.
		function newsItemClick(i) {
			var url = '/Matrix/home/NewsModalContent' + '?selectedIndex=' + i;
			if (!$("#mtrxNewsModal").hasClass("show")) {
				//$('#mtrxNewsModal').modal({keyboard: false, backdrop: 'static'});
				$('#mtrxNewsModal').modal('show');
				mtrxLoading.Show("mtrxNewsModalContent");
			}

			$.ajax({
				url: url,
				type: 'GET',
				success: function (data, status, xhr) {
					$('#mtrxNewsModalContent').html(data);
					mtrxLoading.Hide("mtrxNewsModalContent");
				},
				error: function (data, status, xhr) {
					mtrxForm.Failure(data, status, xhr);
				}
			});
		}

		//****************************************************************************
		//	Handle news item print button click.
		function handleNewsPrint() {
			var strNewsPrintHTML = $('#mtrxNewsModalHeader').html() + $("#NewsMainContent").html();

			if ($('#NewsDetailForPrint').length <= 0) {
				var $newsDetailForPrint = $('<DIV>', {
					'id': 'NewsDetailForPrint',
					'class': 'mtrx-modalPrintHelper',
					'html': strNewsPrintHTML
				});
				$('.mtrx-body').addClass("mtrx-modalNoPrint").append($newsDetailForPrint);

				// Clean up when modal closes.
				$('#newsModal').on('hide.bs.modal', function (e) {
					$newsDetailForPrint.detach();
					$(".mtrx-body").removeClass("mtrx-modalNoPrint");
				});
			}
			else {
				// This condition is when switching news items while the modal is already open.
				$('#NewsDetailForPrint').html(strNewsPrintHTML);
			}

			window.print();
		}

		//****************************************************************************
		//	Public.
		return {
			NewsItemClick: newsItemClick,
			HandleNewsPrint: handleNewsPrint,
			ShowModalFirstTime: showModalFirstTime
		};

	var mtrxHotSheetsWidget = (function () {
		var htmlFieldPrefix = 'widgetHotSheets_';

		function hotsheetsDragDrop(event) {
			mtrxDragDrop.Init("", mtrxHotSheetsWidget.SaveHotsheetsState);

		//****************************************************************************
		//	Handle Date Range option change.
		function handleCustomDateRangeOption() {
			var dateOrTime = $('input[name="widgetHotsheetCustomDateRange"]:checked').val();
			$("#widgetHotsheetDate").toggleClass('d-none', dateOrTime != "1");
			$("#widgetHotsheetTime").toggleClass('d-none', dateOrTime != "2");
		}

		//****************************************************************************
		//	Run Hot Sheet for given ID.
		function runHotsheet(hotsheetID) {
			$("#widgetHotsheetTypes").val(hotsheetID);

			//IF "custom" is selected from the front, flip to show the custom options.
			if ($('#' + htmlFieldPrefix + 'm_ddlPeriod').val() == 6) {
				$('#widgetHotsheet').toggleClass('mtrx-is-flipped');
				$('#widgetHotsheetManageMode').addClass('d-none');
				$('#widgetHotsheetCustomTime').removeClass('d-none');
			}
			else {
				mtrxLoading.Show();
				var runURL = '/Matrix/Widgets/HotSheets/Run';
				window.location.href = window.location.origin + runURL + '?hotsheetID=' + $("#widgetHotsheetTypes").val() + '&periodID=' + $('#' + htmlFieldPrefix + 'm_ddlPeriod').val();
			}
		}

		//****************************************************************************
		//	Validate user input for custom date range and then send it to server for further validation.
		//	If server side validations are passed then run the hotsheet with custom date range,
		//	Otherwise show validation error(s) to user.
		function runCustomHotsheet() {
			var getCustomDateURL = '/Matrix/Widgets/HotSheets/GetCustomDate' + '?hotsheetID=' + $("#widgetHotsheetTypes").val() + '&periodID=' + $('#' + htmlFieldPrefix + 'm_ddlPeriod').val();
			var dateOrTime = $('input[name="widgetHotsheetCustomDateRange"]:checked').val();

			//Gather date or time parameters based on current choice.
			if (dateOrTime == "1") {
				if ($("#" + htmlFieldPrefix + "HotSheetDateRange").val() == "") {
					$("#" + htmlFieldPrefix + "HotSheetDateRange").addClass("input-validation-error");
					return;
				}

				getCustomDateURL += '&date=' + $("#" + htmlFieldPrefix + "HotSheetDateRange").val();
			}
			else {
				if ($("#" + htmlFieldPrefix + "HotSheetFromTime").val() == "") {
					$("#" + htmlFieldPrefix + "HotSheetFromTime").addClass("input-validation-error");
					return;
				}

				var fromAMPM = ($("#" + htmlFieldPrefix + "fromAMPM").is(":checked")) ? "AM" : "PM";
				var toAMPM = ($("#" + htmlFieldPrefix + "toAMPM").is(":checked")) ? "AM" : "PM";
				getCustomDateURL += '&from=' + $("#" + htmlFieldPrefix + "HotSheetFromTime").val()
					+ '&fromAMPM=' + fromAMPM
					+ '&to=' + $("#" + htmlFieldPrefix + "HotSheetToTime").val()
					+ '&toAMPM=' + toAMPM;
			}

			mtrxLoading.Show();
			$.get(window.location.origin + getCustomDateURL, null, function (data) {
				var data = JSON.parse(data);
				if (data.success) {
					var runURL = '/Matrix/Widgets/HotSheets/Run';
					window.location.href = window.location.origin + runURL + '?hotsheetID=' + $("#widgetHotsheetTypes").val() + '&periodID=' + $('#' + htmlFieldPrefix + 'm_ddlPeriod').val() + '&date=' + encodeURIComponent(data.date);
				}
				else {
					mtrxLoading.Hide();
					//TODO: use model validation?
					if (data.badDate) {
						$("#" + htmlFieldPrefix + "HotSheetDateRange").addClass("input-validation-error");
					}
					if (data.badFrom) {
						$("#" + htmlFieldPrefix + "HotSheetFromTime").addClass("input-validation-error");
					}
					if (data.badTo) {
						$("#" + htmlFieldPrefix + "HotSheetToTime").addClass("input-validation-error");
					}
					if (data.alert) {
						$error = $('<div>', { 'class': 'alert alert-danger mtrx-alert--withIcon alert-dismissible fade show', 'role': 'alert' });
						$error.html('<i class="h2 far fa-exclamation-circle"></i>' + data.alert + '<button type="button" class="btn-close" data-bs-dismiss="alert" aria-label="Close"></button>');
						$("#widgetHotsheet .mtrx-flipper-back .card-body").prepend($error);
					}
				}
			});
		}

		function saveHotsheetsState() {

			var count = 0;

			$("[id=widgetHotSheets_selectedList] li").each(function()

				success: (function(partialViewResult) {
					$("#widgetHotsheet").html(partialViewResult);
					$('#widgetHotsheetCustomTime').addClass('d-none');
				})
			});
		}

	var mtrxMyListingsWidget = (function () {
		var isDisplayedAll = false;
		var maxDisplayedQueries = 8;
		var initialSelectedIds;

		function refresh() {
			reloadWidget(isDisplayedAll, true);
		}

		function toggleDisplayAll() {
			reloadWidget(!isDisplayedAll, false);
		}

		function updateSettingsState() {
			var checkboxes = $("#widgetMyListingsSettings input[type='checkbox']");
			if (checkboxes.filter(":checked").length < maxDisplayedQueries)
			{
				checkboxes.removeAttr("disabled");
			}
			else
			{
				checkboxes.not(":checked").attr("disabled", true);
			}
		}

		function flipToSettings() {
			flip();
			updateSettingsState();
			initialSelectedIds = getSelectedIds();
		}

		function flipToList() {
			var currentSelectedIds = getSelectedIds();
			if (areIdArraysEqual(initialSelectedIds, currentSelectedIds)) {
				flip();
			}
			else {
				$.ajax({
					type: 'POST',
					url: '/Matrix/Widgets/MyListings/SaveUserQueries',
					data: { queriesIds: currentSelectedIds },
					beforeSend: mtrxLoading.Show('widgetMyListings'),
					success: function () {
						flip();
						refresh();
					},
					error: function (data, status, xhr) {
						mtrxForm.Failure(data, status, xhr);
					}
				});
			}
		}

		function flip() {
			$('#widgetMyListings').toggleClass('mtrx-is-flipped')
		}

		function reloadWidget(displayAll, ignoreCache) {
			$.ajax({
				type: 'GET',
				url: '/Matrix/Widgets/MyListings/Reload',
				data: { displayAll: displayAll, ignoreCache: ignoreCache },
				beforeSend: mtrxLoading.Show('widgetMyListings'),
				success: function (data) {
					mtrxLoading.Hide('widgetMyListings');
					$('#widgetMyListings').html(data);
					mtrxWidgets.HandleDragDropSetup($('#widgetMyListings').find(".mtrx-card-header-dragdrop-icon"));
					if (displayAll){
						mtrxWidgets.AutoResizeWidgetHeight('widget_MyListings');
					}
					else{
						mtrxWidgets.ForceWidgetResize('widget_MyListings');
					}
				},
				error: function (data, status, xhr) {
					mtrxForm.Failure(data, status, xhr);
				}
			});
		}

		function getSelectedIds() {
			var selectedIds = [];
			$("#widgetMyListingsSettings input[type='checkbox']:checked")
				.each(function () {
					var idParts = this.id.split("_");
					selectedIds.push(idParts[1]);
				});

			return selectedIds;
		}

		function areIdArraysEqual(ids1, ids2) {
			if (ids1.length != ids2.length) {
				return false;
			}

			var i = ids1.length;
			while (i--) {
				if (ids1[i] !== ids2[i]) {
					return false;
				}
			}

			return true;
		}


		//****************************************************************************
		//	Public.
		return {
			Refresh: refresh,
			ToggleDisplayAll: toggleDisplayAll,
			FlipToSettings: flipToSettings,
			FlipToList: flipToList,
			UpdateSettingsState: updateSettingsState,
		};
	})();

	//********************************************************************************
	//	Market Watch
	//
	//	Functions for reloading Market Watch List and resetting selected Market Watch.
	//********************************************************************************
	var mtrxMarketWatchWidget = (function () {
		var htmlFieldPrefix = 'widgetMarketWatch_';

		//****************************************************************************
		//	Get Market Watch list as per the options selected by user.
		function reloadList(ignoreCache) {
			var dataToSend = {
				marketWatchId: $("#" + htmlFieldPrefix + "SelectedMarketWatchID").val(),
				periodId: $("#" + htmlFieldPrefix + "SelectedPeriod_ID").val(),
				ignoreCache: ignoreCache,
				__RequestVerificationToken: $("[name='__RequestVerificationToken']").val()
			};

			mtrxLoading.Show("widgetMarketWatch");
			$.ajax({
				url: '/Matrix/Widgets/MarketWatch/GetMarketWatchList',
				type: 'POST',
				data: dataToSend,
				success: (function (result) {
					mtrxLoading.Hide("widgetMarketWatch");
					$("#widgetMarketWatch").html(result.partial);
					mtrxWidgets.HandleDragDropSetup($('#widgetMarketWatch').find(".mtrx-card-header-dragdrop-icon"));
				})
			});
		}

		//****************************************************************************
		//	Reset selected Market Watch.
		function reset() {
			var dataToSend = {
				marketWatchId: $("#" + htmlFieldPrefix + "SelectedMarketWatchID").val(),
				__RequestVerificationToken: $("[name='__RequestVerificationToken']").val()
			};

			mtrxLoading.Show("widgetMarketWatch");
			$.ajax({
				url: '/Matrix/Widgets/MarketWatch/Reset',
				type: 'POST',
				data: dataToSend,
				success: (function (result) {
					mtrxLoading.Hide("widgetMarketWatch");
					$("#widgetMarketWatch").html(result.partial);
					mtrxWidgets.HandleDragDropSetup($('#widgetMarketWatch').find(".mtrx-card-header-dragdrop-icon"));
				})
			});
		}
		//****************************************************************************
		//	Public.
		return {
			ReloadList: reloadList,
			Reset: reset
		};
	})();


								
								
									
	//********************************************************************************
	//	Portal Visitors

	//	Function to open Portal URL in different window.
	//********************************************************************************
	var mtrxPortalVisitorsWidget = (function () {
		var OpenWindow_Wins = {};
		function openWindow(strURL, strTarget, strFeatures) {
			var w = OpenWindow_Wins[strTarget];
			if (w && !w.closed) {
				w.close();
				OpenWindow_Wins[strTarget] = null;
			}
			w = window.open(strURL, strTarget, strFeatures);
			w.focus();
			OpenWindow_Wins[strTarget] = w;
		}
		//****************************************************************************
		//	Public.
		return {
			OpenWindow: openWindow
		};
	})();
