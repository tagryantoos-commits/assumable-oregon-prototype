                            (function () {
                              function getResolution() {
                                // If you truly want *screen* resolution, use screen.width/height.
                                // If you want *viewport*, keep innerWidth/innerHeight.
                                var vw = window.innerWidth || document.documentElement.clientWidth || 0;
                                var vh = window.innerHeight || document.documentElement.clientHeight || 0;
                                return (vw && vh) ? (vw + 'x' + vh) : 'Unknown';
                              }

                              function setResolution() {
                                try {
                                  var res = getResolution();

                                  // cookie (for next request)
                                  document.cookie = 'ScreenResolution=' + encodeURIComponent(res) + '; path=/; samesite=lax';

                                  // hidden field (for postback)
                                  var form = document.querySelector('form');
                                  if (!form) return;

                                  var hf = document.getElementById('__ScreenResolution');
                                  if (!hf) {
                                    hf = document.createElement('input');
                                    hf.type = 'hidden';
                                    hf.id = '__ScreenResolution';
                                    hf.name = '__ScreenResolution'; // IMPORTANT: match what server reads
                                    form.appendChild(hf);
                                  }
                                  hf.value = res;
                                } catch (e) { /* ignore */ }
                              }

                              // Run after DOM is ready (form exists)
                              if (document.readyState === 'loading') {
                                document.addEventListener('DOMContentLoaded', setResolution);
                              } else {
                                setResolution();
                              }

                              // Ensure it’s set right before any submit
                              document.addEventListener('submit', function () { setResolution(); }, true);

                              // Optional: keep it current
                              window.addEventListener('resize', setResolution);
                            })();
